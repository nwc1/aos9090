#include<stdio.h>
#include<signal.h>
#include<stdlib.h>
void sighup();
void sigint();
void sigquit();
main()
{
int pid,i,j,k;
if ((pid=fork())<0)
{
perror("fork");
exit(1);
}
if(pid==0)
{
signal(SIGHUP,sighup);
signal(SIGINT,sigint);
signal(SIGQUIT,sigquit);
for(;;);
}
else
{
j=0;
for(i=1;i<=5;i++)
{
j++;
printf("PARENT: sending SIGHUP signal:%d\n",j);
kill(pid,SIGHUP);
sleep(3);
printf("PARENT:sending SIGHUP signal:%d\n",j);
kill(pid,SIGINT);
sleep(3);
}
sleep(3);
printf("PARENT:sending SIGQUIT");
kill(pid,SIGQUIT);
}
}
void sighup()
{
signal(SIGHUP,sighup);
printf("child:i have received sighup\n");
}
